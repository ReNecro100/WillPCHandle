-- 1. Таблица GPU
CREATE TABLE GPU (
    id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    RAM TEXT
);

-- 2. Таблица CPU
CREATE TABLE CPU (
    id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    Cores INTEGER,
    Speed TEXT
);

-- 3. Таблица DirectX
CREATE TABLE DirectX (
    id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL
);

-- 4. Таблица OS
CREATE TABLE OS (
	id INTEGER PRIMARY KEY,
	Name TEXT NOT NULL
);

-- 5. Таблица Config
CREATE TABLE Config (
    id INTEGER PRIMARY KEY,
    GPU INTEGER,
    CPU INTEGER,
    RAM TEXT,
    OS INTEGER,
    DirectX INTEGER,
    FOREIGN KEY (GPU) REFERENCES GPU(id),
    FOREIGN KEY (CPU) REFERENCES CPU(id),
    FOREIGN KEY (DirectX) REFERENCES DirectX(id),
    FOREIGN KEY (OS) REFERENCES OS(id)
);

-- 6. Таблица HDD
CREATE TABLE HDD (
    id INTEGER PRIMARY KEY,
    Config INTEGER NOT NULL,
    Name TEXT NOT NULL,
    FreeSpace TEXT,
    FOREIGN KEY (Config) REFERENCES Config(id)
);

-- 7. Таблица User
CREATE TABLE [User] (
    id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    Password TEXT NOT NULL,
    SteamLink TEXT,
    Config INTEGER,
    FOREIGN KEY (Config) REFERENCES Config(id)
);

-- 8. Таблица ApptotalInfo
CREATE TABLE ApptotalInfo (
    id INTEGER PRIMARY KEY,
    [User] INTEGER NOT NULL,
    HeaderImage TEXT,
    Name TEXT NOT NULL,
    Description TEXT,
    Genre TEXT,
    MinimalRequirements INTEGER,
    RecommendedRequirements INTEGER,
    CompatibilityIndicator TEXT,
    FOREIGN KEY ([User]) REFERENCES [User](id)
);

-- 9. Таблица Screenshots
CREATE TABLE Screenshots (
    id INTEGER PRIMARY KEY,
    SteamGame INTEGER NOT NULL,
    [Path] TEXT NOT NULL,
    FOREIGN KEY (SteamGame) REFERENCES ApptotalInfo(id)
);
